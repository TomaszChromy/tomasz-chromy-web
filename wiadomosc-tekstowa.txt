  slug: string;

  content: string;

const BLOG_ROOT_DIR = path.join(process.cwd(), "content", "blog");

/**
 * Zwraca listÄ™ metadanych wszystkich postÃ³w w danym jÄ™zyku
 * (bez treÅ›ci â€“ idealne do listy /blog, paginacji, â€related postsâ€).

export function getAllPosts(lang: "pl" | "en" | "nl" = "pl"): BlogPostMeta[] {
  const dir = getLangDir(lang);

  if (!fs.existsSync(dir)) {
    return [];
  }

  const files = fs
    .readdirSync(dir)
    .filter((file) => file.endsWith(".md") || file.endsWith(".mdx"));

  const posts: BlogPostMeta[] = files.map((file) => {
    const fullPath = path.join(dir, file);
    const fileContents = fs.readFileSync(fullPath, "utf8");
    const { data } = matter(fileContents);

    if (!data.slug) {
      throw new Error(`Brak pola "slug" w pliku: ${fullPath}`);
    }

    return {
      slug: data.slug,
      title: data.title,
      excerpt: data.excerpt,
      date: data.date,
      readingTime: data.readingTime,
      category: data.category,
      tags: data.tags ?? [],
      language: data.language ?? lang,
      featured: data.featured ?? false,
    } as BlogPostMeta;
  });

  // Nowsze posty wyÅ¼ej
  return posts.sort((a, b) => (a.date < b.date ? 1 : -1));
}

/**

export function getPostBySlug(
  slug: string,
  lang: "pl" | "en" | "nl" = "pl",
): BlogPost | null {
  const dir = getLangDir(lang);

  if (!fs.existsSync(dir)) {
    return null;
  }

  const files = fs
    .readdirSync(dir)
    .filter((file) => file.endsWith(".md") || file.endsWith(".mdx"));

  for (const file of files) {
    const fullPath = path.join(dir, file);
    const fileContents = fs.readFileSync(fullPath, "utf8");
    const { data, content } = matter(fileContents);

    if (data.slug === slug) {
      return {
        slug: data.slug,
        title: data.title,
        excerpt: data.excerpt,
        date: data.date,
        readingTime: data.readingTime,
        category: data.category,
        tags: data.tags ?? [],
        language: data.language ?? lang,
        featured: data.featured ?? false,
        content,
      } as BlogPost;
    }
  }

  return null;
}
